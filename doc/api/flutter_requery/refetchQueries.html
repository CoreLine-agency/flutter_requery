void refetchQueries(BuildContext context, List<dynamic> keys) {
  List<String> cacheKeys = keys.map((k) => _transformKey(k)).toList();
  _QueryNotifier provider = Provider.of<_QueryNotifier>(context, listen: false);
  cacheKeys.forEach((k) {
    provider.refetch(k);
  });
}
